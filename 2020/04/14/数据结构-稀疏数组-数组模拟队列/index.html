数据结构-稀疏数组及数组模拟队列

稀疏数组和数组模拟队列:

稀疏数组

基本概述

当一个二维数组大部分数据都为0的时候，可以用到稀疏数组来保存该数组。

稀疏数组的第一行分别存放原二维数组的行数，列数，以及非0的值个数

从第二行开始就是存放非0数据分别在哪一行哪一列….

二维数组与稀疏数组的转换

代码如下：

package com.yjr.sparseArray;

import java.io.*;

public class sparseArray {

 public static void main(String args[]) {

     //原二维数组转稀疏数组
     //假设定义一个二维数组，11行，11列，非0数据为sum个。

     int [][] chessArr=new  int [11] [11];
     chessArr[1][3]=8;
     chessArr[2][4]=7;
     chessArr[1][6]=11;
     chessArr[6][3]=5;
     chessArr[8][9]=15;


     //怎么转？思路如下
     //首先把二维数组进行遍历，得到有效数据个数
     //定义一个sum来记录原二维数组的非0个数
     int sum=0;
     System.out.println("遍历原二维数组.....");

     for (int [] rows:chessArr){

         //遍历行数组出来
         //再遍历行数组里面的每一个数据
         for (int data:rows){

             if(data!=0){

               sum++;

               //输出一下每一个行数组的数据
             }
             System.out.printf("%d\t",data);


         }
         System.out.println();

     }
     System.out.println("有效数据为:" + sum);


     //现在要把该二维数组转成稀疏数组
     //定义稀疏数组
     int [] [] sparseArray=new int [sum+1][3];

     sparseArray[0][0]=11;
     sparseArray[0][1]=11;
     sparseArray[0][2]=sum;

     //现在要对稀疏数组进行赋值

     //遍历原二维数组
     //定义一个count来记录是第几个非0数据
     int count=0;
     for (int i=0;i<11;i++){

         for (int j=0;j<11;j++){

             if(chessArr[i][j]!=0){

                 count++;

                 sparseArray[count][0]=i;
                 sparseArray[count][1]=j;
                 sparseArray[count][2]=chessArr[i][j];



             }


         }


     }

     //将稀疏数组写入文件中

     try {

         FileWriter writer=new FileWriter("map.txt");
         BufferedWriter bufferedWriter=new BufferedWriter(writer);

         for (int [] rows:sparseArray){

             for (int data:rows){

                 bufferedWriter.write(String.valueOf(data)+"\t");
             }
             bufferedWriter.newLine();

         }
         bufferedWriter.flush();
         bufferedWriter.close();
         writer.close();

     } catch (IOException e) {
         e.printStackTrace();
     }

     System.out.println("输出稀疏数组!");
     //输出该稀疏数组
     for (int a=0;a<sparseArray.length;a++){

         System.out.printf("%d\t%d\t%d\t",sparseArray[a][0],sparseArray[a][1],sparseArray[a][2]);
         System.out.println();
     }

     //先把稀疏数组从文件中取出来

     int [] [] spareArr2=new int[sum+1][3];
     try {
         FileReader reader=new FileReader("map.txt");
         BufferedReader bufferedReader=new BufferedReader(reader);



         String[] s1 = null;
         int[] spa=null;
         String s ;
         int account=0;
         while ((s=bufferedReader.readLine())!=null){

             s1=s.split("\t");

          spa=new int[s1.length];
            for (int i=0;i<s1.length;i++){

               spa[i]=Integer.parseInt(s1[i]);
           }

            for(int i=0;i<spa.length;i++){

                spareArr2[account][i]=spa[i];

            }
            account++;

        }

     } catch (Exception e) {
         e.printStackTrace();
     }
     //遍历一下读取出来的稀疏数组
     System.out.println("遍历读取出来的稀疏数组");
     for (int[] rows:spareArr2){

         for (int data:rows){

             System.out.printf("%d\t",data);
         }
         System.out.println();
     }

     //接下来是讲稀疏数组恢复成原始的二维数组

     //首先明确恢复成的二维数组是多少行，多少列
     int chessArr2[][]=new int[sparseArray[0][0]][sparseArray[0][1]];

     //接下来对该数组进行赋值
     //遍历稀疏数组
   for(int i=1;i<sparseArray.length;i++){

       chessArr2[sparseArray[i][0]][sparseArray[i][1]]=sparseArray[i][2];

   }

   //输出该二维数组
     System.out.println("输出从稀疏数组恢复的二维数组");
     for (int [] rows:chessArr2){

         for (int data:rows){
             System.out.printf("%d\t",data);
         }
         System.out.println();
     }


 }




 }

数组模拟队列

非环形队列

front:初始值为-1，头元素的前一个位置

rear:初始值为-1,指向尾元素

maxSize:数组容量

队空：front==rear

队满: rear=maxSize-1

实现的代码

package com.yjr.queue;



import java.util.Scanner;

public class ArrayQueueDemo {


//利用数组模拟队列


public static void main(String[] args) {

    //创建一个队列

    ArrayQueue arrayQueue = new ArrayQueue(5);

    Scanner scanner = new Scanner(System.in);

    //菜单
    boolean top=true;
    while (top) {

        System.out.println("s(show): 显示队列");
        System.out.println("e(exit): 退出程序");
        System.out.println("a(add): 添加数据到队列");
        System.out.println("g(get): 从队列取出数据");
        System.out.println("h(head): 查看队列头的数据");

        //接收一个字符
        char key = scanner.next().charAt(0);


        switch (key) {

            case 's':

                try {
                    arrayQueue.showQueue();
                    System.out.println();
                } catch (Exception e) {
                    e.printStackTrace();
                    // System.out.println(e.getMessage());
                }
                break;

            case 'e':
                scanner.close();
                System.out.println("程序已退出!");
                top=false;
                break;

            case 'a':


                System.out.println("请输入要添加的数据");
                int n = scanner.nextInt();
                try {
                    arrayQueue.addQueue(n);
                } catch (Exception e) {
                    e.printStackTrace();
                }
                break;

            case 'g':
                try {
                   int a= arrayQueue.getQueue();
                    System.out.printf("取出的数据为%d\n",a);
                } catch (Exception e) {
                    e.printStackTrace();
                }
                break;

            case 'h':

                try {
                    arrayQueue.headQueue();
                } catch (Exception e) {
                    e.printStackTrace();
                }
                break;

            default:

                System.out.println("命令输入错误！");
                break;

        }


    }


}


}


class  ArrayQueue{

//这是数组队列类

private int MaxSize; //数组的最大容量
private int front; //指向队头元素的前一个，默认为-1
private int rear;  //指向队尾元素，默认为-1
private int [] arr; //数组

public ArrayQueue(int maxSize) {
    MaxSize=maxSize;
    arr=new int[MaxSize];
    front=-1;
    rear=-1;

    //这样通过构造函数就可以创建指定大小的数组
}


//写方法

//判断是否为空

public boolean isEmpty(){

    return  front==rear;//两者相等的话就说明为空
}

//判断是否为满

public  boolean isFull(){

    return  rear==MaxSize-1;
}

//添加队列
public  void  addQueue(int n){

    //判断是否为满
    if(isFull()){

        System.out.println("队列已满，添加失败！");
        return;
    }

    //否则添加队列
    //rear向后移
    rear++;
    arr[rear]=n;
    System.out.println("添加成功了！");
}

//取队列的数据(从头取)
public  int getQueue(){

    //判断是否空
    if(isEmpty()){

        throw new RuntimeException("队列空，不能取数据");

    }
    //否则
    //front后移
    front++;
    //返回数据
    return  arr[front];

}

//显示队列的所有数据
public void showQueue(){

    //判断是否空
    if (isEmpty()){

        System.out.println("队列为空，无法取出");

    }

    for(int i=0;i<arr.length;i++){

        System.out.printf("arr[%d]=%d\t",i,arr[i]);

    }

}

//查看队列的头元素
public  void headQueue(){
    if (isEmpty()) {
        System.out.println("队列为空，查看头信息失败！");
    }
    front++;
    System.out.println(arr[front]);

}


}

环形队列

front:初始值 0，指向头元素

rear:初始值 0，指向尾元素的后一位

队空: front==rear

队满：(rear+1)%maxSize==front

有效数据个数: (rear + maxSize - front)% maxSize

添加队列完后:rear =(rear+1)%maxSize

取出队列元素后: front=(front+1)%maxSize

取余是为了防止越界

实现代码

package com.yjr.queue;


import java.util.Scanner;

public class CricleArrayQueueDemo {

public static void main(String args[]) {

    //创建一个队列

    CricleArrayQueue cricleArrayQueue = new CricleArrayQueue(5);

    Scanner scanner = new Scanner(System.in);

    //菜单
    boolean top = true;
    while (top) {

        System.out.println("s(show): 显示队列");
        System.out.println("e(exit): 退出程序");

        System.out.println("g(get): 从队列取出数据");
        System.out.println("h(head): 查看队列头的数据");

        //接收一个字符
        char key = scanner.next().charAt(0);


        switch (key) {

            case 's':

                try {
                    cricleArrayQueue.showQueue();

                }
                break;

            case 'e':

                System.out.println("程序已退出!");
                top = false;
                break;

            case 'a':

                int n = scanner.nextInt();
                try {
                    cricleArrayQueue.addQueue(n);
                } catch (Exception e) {
                    e.printStackTrace();

            case 'g':

                }

                } catch (Exception e) {

  class   CricleArrayQueue{

    //这是数组队列类

    private int MaxSize; //数组的最大容量
    private int front; //指向队头元素的前一个，默认为-1
    private int rear;  //指向队尾元素，默认为-1
    private int [] arr; //数组

    public  CricleArrayQueue(int maxSize) {
        MaxSize=maxSize;
        arr=new int[MaxSize];
        front=0;
        rear=0;

        //这样通过构造函数就可以创建指定大小的数组
    }


    //写方法

    //判断是否为空

    public boolean isEmpty(){

        return  front==rear;//两者相等的话就说明为空
    }

    //判断是否为满

    public  boolean isFull(){

        return  (rear+1)%MaxSize==front; //环形对列的满条件
    }

    //添加队列
    public  void  addQueue(int n){

        //判断是否为满
        if(isFull()){

            System.out.println("队列已满，添加失败！");
            return;
        }

        //否则添加队列
        //rear向后移
        arr[rear]=n;

        rear=(rear+1)%MaxSize;
        System.out.println("添加成功了！");
    }

    //取队列的数据(从头取)
    public  int getQueue(){

        //判断是否空
        if(isEmpty()){

            throw new RuntimeException("队列空，不能取数据");

        }
        //否则
        //front后移
        int a=arr[front];

        front=(front+1)%MaxSize;
        //返回数据
        return  a;

    }

    //显示队列的所有数据
    public void showQueue(){

        //判断是否空
        if (isEmpty()){

            System.out.println("队列为空，无法取出");

        }

        for(int i=front;i<front+size();i++){

            System.out.printf("arr[%d]=%d\t",i%MaxSize,arr[i%MaxSize]);

        }

    }

    //查看队列的头元素
    public  void headQueue(){
        if (isEmpty()) {
            System.out.println("队列为空，查看头信息失败！");
        }

        System.out.println(arr[front]);

    }

    public  int size(){

        //对列有效数据个数
        return  (rear-front+MaxSize)%MaxSize;
    }


}
