数据结构-查找算法第一辑

查找算法：

  1. 顺序(线性)查找
  2. 二分查找/折半查找

线性（顺序）查找算法:

代码如下：

   public class SeqSearch {


//顺序查找


 public static void main(String args[]) {

     int arr[]={1,3,5,7,9,2,4};
     int i = seqSerach(arr, 90);
     System.out.println(i);

 }


 public static   int seqSerach(int arr[],int findVal){


     //遍历数组

     for (int i=0;i<arr.length;i++){

         if(arr[i]==findVal){

             return  i;
         }


     }
     return  -1;

 }

}

二分查找

代码如下:

  public class BinarySearch {


 public static void main(String args[]) {

     //二分查找(顺序表)

     int [] arr={1,3,3,3,3,7,9,11,13};

 // System.out.println( binarySearch(arr,0,arr.length-1,1));

     ArrayList<Integer> arrayList = binarySearch2(arr, 0, arr.length - 1, 3);

     System.out.println(arrayList);


 }


 //先写一个只能查到单个元素的索引的方法，重复元素的查不到
 public static int  binarySearch(int [] arr,int left,int right,int findVal) {


     //遍历数组
     //既然是递归，那么就要有终止条件，找到了就返回这是一个，找不到的话也要设置一个条件
     if (left > right) {

         return -1;
     }
     int mid = (left + right) / 2;//中间的指针


         if (findVal > arr[mid]) {

             //说明要寻找的值比中间值的指针大，那么就要向右边继续寻找，也就是向右递归。
             return binarySearch(arr, mid + 1, right, findVal);

         } else if (findVal < arr[mid]) {

             //否则就是要寻找的值比中间值小，向左递归
             return binarySearch(arr, left, mid - 1, findVal);

         } else {

             //否则就是恰好中间的值就是
             return mid;

         }





 }



 //现在编写一个可以查询重复元素的索引的方法。用集合来接收

public static ArrayList<Integer> binarySearch2(int [] arr,int left,int right,int findVal){



     if(left>right){

       return new ArrayList<Integer>();
     }

     int mid=(left+right)/2;
     if(findVal>arr[mid]){

        return binarySearch2(arr,mid+1,right,findVal);

     }
     else if(findVal<arr[mid]){

          return  binarySearch2(arr,left,mid-1,findVal);
     }
     else{

        /*

        这种方法可行的。
        下面的没注释的也是可行的
         //这个环节说明是到了找到到第一个元素的时候了，但是不要立即返回。
         //如果这个时候left还没有等于right,就说明可能还有相同数据,继续寻找
         ArrayList<Integer> arrayList=new ArrayList<>();

         if(left==right){

             return  arrayList;
         }
         for(int i=0;i<arr.length;i++){

             if(arr[i]==arr[mid]){

                 //就把数据放进集合中并返回
                 arrayList.add(i);

             }

         }
       return   arrayList;

         */

        ArrayList<Integer> arrayList=new ArrayList<>();

        int temp=mid-1;  //定义一个向左边扫描的指针

         while (true){

             //说明不满足加入集合中的条件
             if(temp<0||arr[temp]!=findVal) {
                 break;
             }
             //为什么这样设计呢，因为数组它是有序的，假如你的数据有重复的话，那么你的当前找到的数据的指针(进来else的指针)如果你的左边都没了
             // ，那么就没必要找了，因为根本不可能再有了，右边也是同样的道理
             arrayList.add(temp);
             temp--;

         }

         //这里是把刚进来else就找到的数据放进集合里
         arrayList.add(mid);

         temp =mid+1;
         while (true){

             if(temp<0||arr[temp]!=findVal){

                 break;
             }

             arrayList.add(temp);
             temp++;

         }

         return  arrayList;
     }



}


}
