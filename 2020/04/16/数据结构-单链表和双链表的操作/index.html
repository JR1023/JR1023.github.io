public void add(HeroNode heroNode){

    //得遍历链表，找到最后一个节点就退出,在节点的后面添加
    //需要一个辅助变量来辅助遍历,头结点不能动

        temp=temp.next;

    }

    //此时temp指向最后一个节点
    temp.next=heroNode;

 //接下来是编号排序（从一开始）且编号唯一的添加

public  void  addOrder(HeroNode heroNode){

    //条件分析：
    /*
    插入：1.节点后面插入(无论链表是空还是不空只需要temp.next==null，退出循环即可实现该方式的插入)
          2.链表中间插入(temp.next.no > 传入节点.no,即可实现从小到大插入)
    不插入:temp.next.no == 传入节点.no,实现唯一插入
   三个条件只要满足一个就会退出循环，实现插与不插
    每次插入节点都要遍历链表，如果遍历到的temp都不满足三种条件，那就后移temp,总有一个temp会满足条件。

     */

    HeroNode temp=head; //添加节点与head要联系起来,况且中间插入需要temp指定在要被插入位置的前一位
    boolean flag=false; //默认插入的节点不重复
    while (true){

        if(temp.next==null){

            //节点后面插入(无论链表是空还是不空只需要temp.next==null，退出循环即可实现该方式的插入)
            break;
        }
        if (temp.next.no >heroNode.no){
            //中间插入需要temp指定在要被插入位置的前一位
            //2.链表中间插入(temp.next.no > 传入节点.no,即可实现从小到大插入)
            break;
        }
        if (temp.next.no==heroNode.no){

            //不插入:temp.next.no == 传入节点.no,实现唯一插入
            flag=true;//重复数据
            break;

        }

        //如果遍历到的temp都不满足三种条件，那就后移temp,总有一个temp会满足条件。
        temp = temp.next;


    }

    //三个条件只要满足一个就会退出循环，实现插与不插
    if(flag){

        //重复数据不插入
        System.out.println("插入的节点数据重复，无法添加！"+heroNode.no);

    }
    else {

        //进行插入操作

        //这里一定要注意，顺序不能颠倒。如果颠倒了那么链表就连不起来了。
        heroNode.next=temp.next;
        temp.next=heroNode;

    }




}

public  void  delete(HeroNode heroNode){

    //遍历找到要被删除节点的前一个节点，这样能把被删除数据前一个与被删除数据后一个的连接连起来
    //所以temp从head开始
    HeroNode temp=head;

    //判断链表是否为空
    if(head.next==null){
        return;
    }
    while (true){

            //说明该temp是被删除节点的前一个节点

            //这是删除的节点不是最后一个节点的时候

                temp.next = temp.next.next;


            break;

        }

        if(temp.next!=null){

            temp=temp.next;
        }
       else {
            System.out.println("未找到要被删除的数据！");
            break;
        }


    }


}

腾讯面试题,反转一个单链表

   public void reverseSingleLinked(){

    //思路：
    /*

    创建一个新的单链表头节点，用来辅助生成反转链表
    根据原单链表的头节点，先遍历出每一个有效节点。(辅助变量从第一个有效节点开始指)
    遍历到每一个新的节点，就把让头r指向它。但是单单这样是不行的，因为这样只是永远在换r的next的指向值
    所以，我们要想一个方法，让新节点进来r.next的同时，也要把以前遍历的节点连在一起

    如 h  1 2 3 4
       r
    当前的cur假设是2

    这个时候呢，当我把r.next指向2的时候，我的1呢？1的话可以这样：

    在cur等于1的时候，让cur指向r(cur.next=r.next),再把r->1(r.next=cur),那么就会形成

    当前cur指向的就是1(第一步循环就是得到r的next域是1)

    然后 再当cur=2时，继续cur.next=r.next,这样2和1就可以连起来了，

    然后再用r.next指向2（r.next=cur）,就会形成 r->2->1的结果，

    以此类推，当3进来的时候,把3指向r的next域（变成2了），这个时候就是3->2->1连起来了，再r->3(r.next-cur),以此类推，最终就可以连起来

    cur.next=r.next  //先把当前cur指向r的next域，那么节点1在节点2


    */

    //实现

    HeroNode cur=head.next;//辅助变量从第一个有效节点开始指)

    HeroNode next=null;//用来后移cur的指向,达到遍历的效果

    HeroNode reverseHead=new HeroNode(0,"","");

    if(head.next==null||head.next.next==null){

        //空的或者只有一个节点，不用反转
        return;

    }
    while (true){

        if(cur==null){

            //说明到最后的节点了
            break;
        }
        //否则
        next=cur.next; //记录下一个cur的指向值

        cur.next=reverseHead.next;

        reverseHead.next=cur;
        cur=next; //后移



    }

    //退出循环后，因为你是新建了一个头，所以你的头结点地址和原来的头是不一样的
    //所以把旧头指向新头的next即可

    head.next=reverseHead.next;

    //遍历一下验证
    list();


}

class  ListNode{

int val;
ListNode next;

public ListNode getNext() {
    return next;
}

@Override
public String toString() {
    return "ListNode{" +
            "val=" + val +
            ", next=" + next +
            '}';
}

public void setNext(ListNode next) {
    this.next = next;
}

public ListNode(int val) {
    this.val = val;
}

/*
好比打架，因为原来两个链表都是默认从小到大排序，也就是，每一组都是从弱到强
动脑想想
要想获得所有人的排名（从弱到强）
首先得拿第一组最弱的和第二组最弱的比，谁输了，也就是谁小，就上排行位
然后把输的那一组的次弱的拿出来，和第二组（上一轮赢的那一组）进行比较，输的上排行榜，以此类推，
到了某一组的人物是null, 另一组有人，说明没对手，直接上排行版
*/

public   static ListNode mergeLinkedList(ListNode l1,ListNode l2){

    if(l1==null){

        return l2;
    }
    if(l2==null){

        return  l1;
    }

    ListNode head=null;

    if(l1.val<=l2.val){

        head=l1;
        head.next=mergeLinkedList(l1.next,l2);

    }
    else {

        head=l2;
        head.next=mergeLinkedList(l1,l2.next);

    }

    return  head;
}
}

双链表

管理单向链表的缺点分析:

  1. 单向链表，查找的方向只能是一个方向，而双向链表可以向前或者向后查找。

  2. 单向链表不能自我删除，需要靠辅助节点 ，而双向链表，则可以自我删除，所以前面我们单链表删除时节点，总是找到 temp,temp 是待删除节点的前一个节点。
