骑士周游(马踏棋盘游戏的解法)

马踏棋盘算法

  1. 马踏棋盘算法也被称为骑士周游问题

  2. 将马随机放在国际象棋的 8×8 棋盘 Board[0～7][0～7]的某个方格中，马按走棋规则(马走日字)进行移动。要求 每个方格只进入一次，走遍棋盘上全部 64 个方格

  3. 游戏演示: http://www.4399.com/flash/146267_2.htm

  1. 马踏棋盘问题(骑士周游问题)实际上是图的深度优先搜索(DFS)的应用。

  2. 如果使用回溯（就是深度优先搜索）来解决，假如马儿踏了 53 个点，如图：走到了第 53 个，坐标（1,0），发 现已经走到尽头，没办法，那就只能回退了，查看其他的路径，就在棋盘上不停的回溯…… ，思路分析+代码 实现

  1. 分析第一种方式的问题，并使用贪心算法（greedyalgorithm）进行优化。解决马踏棋盘问题.

代码如下：

     public class HorseChessboard {

public  static  int X=8; //8列(x坐标)
public  static  int Y=8; //8行（y坐标）
 static   boolean  [] isVisited=new boolean[X*Y]; //记录已经访问的位置
  static boolean finished=false;//是否已经完成

 //马踏棋盘游戏
//问题描述：8*8国际象棋盘中，要求马走走遍棋盘的每一个方格，要怎么走
//马走日字，即只能是先横走或者竖走，然后再从横或者竖的地方向对角线走

/*
这个其实是图的深度优先遍历
就是说先从一个方向一直，不通再回溯去寻找其他

思路分析:
首先是棋盘，需要定义一个二维数组来设计

记录当前节点已经访问，计算马在当前位置还能走的地方(最多有8个位置)，用一个集合来存储，每走一步step+1
直到step等于棋盘的总格子数，遍历集合中存放的可以走的位置，先选择第一个一直递归下去，如果走的通就一直走，否则就回溯
每到一个方法，如果不能再走了，发现step没有达到数量，说明任务没有完成，那么就把当前这个方法的棋盘置为0

 */


 public static void main(String args[]) {

   int row=1;
   int col=1;
   int [] []chess=new int [X][Y];
   horseChessboard(chess,row-1,col-1,1);

   //输出

     for(int [] rows:chess){

         System.out.println(Arrays.toString(rows));
     }
 }


 //马踏棋盘核心方法

//row是当前是第几行，col是当前是第几列(row和col都是从0开始)
public static void  horseChessboard(int [] [] chess,int row,int col,int step ){
  //row就相当于与y坐标，col代表x坐标


    chess[row][col]=step;//每去到一个格子，就把step的值先赋给该格子，也就是初始为0，然后去到下一个格子就step为2了，
                        //然后就把该格子的值置为1，后面会判断是否到达数量
     //设置当前位置已经访问

    isVisited[row * X +col]=true;

    Point cur=new Point(col,row);
    ArrayList<Point> next = next(cur);

    sort(next);
    while (!next.isEmpty()){

        Point point = next.remove(0);

        if(!isVisited[point.y*X + point.x]) {
            horseChessboard(chess, point.y, point.x, step + 1);
        }
    }

    if(step<X*Y && finished==false){

        chess[row][col]=0; //到这说明该格子已经无路可走了，那么就要回溯，所以把该格子的值设为0，这样才不影响

        isVisited[row* X+col]=false;

    }
    else{

        finished=true;
    }



}


 //求得马在当前的位置还能走的位置的集合
 public static  ArrayList<Point> next(Point curPoint){


     ArrayList<Point> ps=new ArrayList<>();

     Point p1=new Point();

     //判断5这个位置
     if((p1.x=curPoint.x-2)>=0 && (p1.y=curPoint.y-1)>=0){

         ps.add(new Point(p1));

     }

     //6这个位置

     if ((p1.x=curPoint.x-1)>=0 && (p1.y=curPoint.y-2)>=0){

         ps.add(new Point(p1));
     }

     //7

     if((p1.x=curPoint.x+1)<X &&(p1.y=curPoint.y-2)>=0){

         ps.add(new Point(p1));
     }

     //0
     if((p1.x=curPoint.x+2)<X &&(p1.y=curPoint.y-1)>=0){

         ps.add(new Point(p1));
     }

     //1
     if((p1.x=curPoint.x+2)<X && (p1.y=curPoint.y+1)<Y){

         ps.add(new Point(p1));

     }
     //2
     if((p1.x=curPoint.x+1)<X && (p1.y=curPoint.y+2)<Y){

         ps.add(new Point(p1));
     }

     //3
     if((p1.x=curPoint.x-1)>=0 && (p1.y=curPoint.y+2)<Y){

         ps.add(new Point(p1));
     }
     //4
     if((p1.x=curPoint.x-2)>=0 &&(p1.y=curPoint.y+1)<Y){

         ps.add(new Point(p1));
     }

 return  ps;



 }


 //运用贪心算法进行优化，每次选择该点可走位置的下一个可走位置的最小的数量的，就是选择一个走的位置，让马走到这个位置（3）后，同时马
//在该位置（3）的下一个可选位置的数量是相比之前选择的其他位置的数量少。
 public static void sort(ArrayList<Point> ps){

     //进行一个非递减排序，也就是升序但是数据可重复
     //比如：1 2 2 3 4 5 5 5 6

     ps.sort(new Comparator<Point>() {
         @Override
         public int compare(Point o1, Point o2) {
             if(next(o1).size()>next(o2).size()){

                 return 1;
             }
             else if(next(o1).size()<next(o2).size()){

                 return -1;
             }
             else {
                 return 0;
             }
         }
     });



 }
}
